SELECT nazwa, ulica
FROM kwiaciarnie
WHERE miasto = 'Malbork'

SELECT id_kwiaciarni, COUNT(id_zam) AS 'liczba zamówień'
FROM `zamowienia` 
GROUP BY id_kwiaciarni;

SELECT kwiaciarnie.nazwa, kwiaciarnie.miasto, zamowienia.kwiaty
FROM kwiaciarnie 
INNER JOIN zamowienia
ON kwiaciarnie.id_kwiaciarni = zamowienia.id_kwiaciarni
WHERE zamowienia.data = '2017-01-07';

CREATE TABLE `kwiaciarnia`.`klienci` 
(`id` INT NOT NULL , 
`imie` VARCHAR(15) NOT NULL , 
`nazwisko` VARCHAR(15),
`rabat` INT, 
PRIMARY KEY (`id`));
